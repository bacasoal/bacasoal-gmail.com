19. Soal Matematika IPA SMA, a > b > 0 dan a^2 + b^2 = 8ab maka nilai a+b/a-b = ....


[tex]a^{2}+b^{2}= 8ab[/tex]

[tex](a+b)^{2}-2ab=8ab[/tex]

[tex](a+b)^{2}=10ab[/tex]

[tex]a+b= \sqrt{10ab} [/tex]


jika : [tex](a-b)^{2}=a^{2}+b^{2}-2ab [/tex]

maka :

[tex]a-b= \sqrt{a^{2}+b^{2}-2ab} [/tex]

[tex]a-b= \sqrt{8ab-2ab} [/tex]

[tex]a-b= \sqrt{6ab} [/tex]   x (-1)

[tex]b-a = - \sqrt{6ab} [/tex]


sehingga :

[tex] \frac{a+b}{b-a}= \frac{ \sqrt{10ab} }{- \sqrt{6ab} } [/tex]

[tex] \frac{a+b}{b-a}=- \sqrt{ \frac{10ab}{6ab} }[/tex]

[tex] \frac{a+b}{b-a}=- \sqrt{ \frac{5}{3} }[/tex]

[tex] \frac{a+b}{b-a}=- \frac{ \sqrt{5} }{ \sqrt{3} } [/tex]

[tex] \frac{a+b}{b-a}=- \frac{ \sqrt{5} }{ \sqrt{3} } . \frac{ \sqrt{3} }{ \sqrt{3} } [/tex]

[tex] \frac{a+b}{b-a}=- \frac{ 1 }{ 3 } \sqrt{15} [/tex]
 
 
QUICK METHOD / CARA CEPAT

[tex] \frac{a+b}{b-a}= \frac{a+b}{-(a-b)}[/tex]

[tex]= -\sqrt{\frac{(a+b)^{2}}{(a-b)^{2}} } [/tex]

[tex]= -\sqrt{\frac{(a^{2}+b^{2})+2ab}{(a^{2}+b^{2})-2ab} } [/tex]

[tex]= -\sqrt{\frac{8ab+2ab}{8ab-2ab} } [/tex]

[tex]= -\sqrt{\frac{10ab}{6ab} }[/tex]

[tex]= -\sqrt{\frac{5}{3} }[/tex]

[tex]=- \frac{1}{3} \sqrt{15} [/tex]
