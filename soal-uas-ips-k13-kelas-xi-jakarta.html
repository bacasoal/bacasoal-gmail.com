[tex]U8 = 3.3^{7} [/tex]
[tex]U8 = 3(2.187)[/tex]

2.
barisan: 6, 18, 54, ... Un
a = 6
r = U3/U2 = 54/18 = 3

[tex]Un = a.r^{n-1}[/tex]
[tex]Un = 6.3^{n-1} [/tex]
[tex]Un = 6.3^{n}.3^{-1} [/tex]
[tex]Un = 6.3^{n}. \frac{1}{3} [/tex]
[tex]Un = 2.3^{n}[/tex]


3.
U3 = 8
U6 = 64

[tex]U3 = a.r^{3-1} [/tex]
[tex]8 = a.r^{2}[/tex]

[tex]U6 = a.r^{6-1} [/tex]

[tex] \frac{U6}{U3} = \frac{64 = a.r^{5}}{8 = a.r^{2}} [/tex]
[tex]8 = r^{3} [/tex]
[tex]2^{3} = r^{3} [/tex]
[tex]r = 2[/tex]
